CREATE TABLE CANDIDATO(
	CEDULA VARCHAR (15) PRIMARY KEY NOT NULL,
	NOMBRE VARCHAR (20),
	LISTA VARCHAR (10),
	MAIL VARCHAR (50),
	ESTADO VARCHAR (1),
	FECHA_CREACION DATETIME DEFAULT GETDATE()
);

SELECT * FROM CANDIDATO;

CREATE TABLE VOTO(
	ID_VOTO INT IDENTITY(0,1) PRIMARY KEY NOT NULL,
	CANDIDATO VARCHAR (15) FOREIGN KEY REFERENCES CANDIDATO(CEDULA),
	ESTADO VARCHAR (1),
	FECHA_CREACION DATETIME DEFAULT GETDATE()
);
